   161	function seededRandom(seed: number) {
   162	    let t = seed += 0x6D2B79F5;
   163	    t = Math.imul(t ^ t >>> 15, t | 1);
   164	    t ^= t + Math.imul(t ^ t >>> 7, t | 61);
   165	    return ((t ^ t >>> 14) >>> 0) / 4294967296;
   166	}

   263	      if (loadedComments.length === 0) {
   264	          const localData = localStorage.getItem(`comments_${itemId}`);
   265	          if (localData) {
   266	            loadedComments = JSON.parse(localData);
   267	          }
   268	      }
   269	      
   270	      // Merge real and fake comments, sort by date
   271	      const allComments = [...loadedComments, ...fakeComments].sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime());
   272	      setComments(allComments);
   273	      setFetching(false);
   274	    };
   275	
   276	    loadComments();
   277	  }, [itemId, fakeComments]);
   278	
   279	  const toggleTag = (tag: string) => {
   280	    setSelectedTags(prev => 
   281	      prev.includes(tag) ? prev.filter(t => t !== tag) : [...prev, tag].slice(0, 3) // Max 3 tags
   282	    );
   283	  };
   284	
   285	  const handleReaction = async (commentId: string, type: 'like' | 'dislike' | 'love') => {
   286	    // Optimistic update
   287	    const updatedComments = comments.map(c => {
   288	      if (c.id === commentId) {
   289	        return { ...c, reactions: { ...c.reactions, [type]: c.reactions[type] + 1 } };
   290	      }
   291	      return c;
   292	    });
   293	    setComments(updatedComments);
   294	    localStorage.setItem(`comments_${itemId}`, JSON.stringify(updatedComments));
   295	
   296	    // Try sending to Google Sheet
   297	    try {
   298	      await fetch(API_ENDPOINT, {
   299	        method: 'POST',
   300	        body: JSON.stringify({ action: 'reactComment', commentId, type }),
   301	      });
   302	    } catch (e) {
   303	      // Ignore errors if script isn't updated
   304	    }
   305	  };
   306	
   307	  const handleSubmit = async (e: React.FormEvent) => {
   308	    e.preventDefault();
   309	    if (!newComment.trim()) return;
   310	
   311	    setLoading(true);
   312	    const commentData: Comment = {
   313	      id: Date.now().toString(),
   314	      itemId,
   315	      author: authorName.trim() || 'Anonymous Wolf',
   316	      text: newComment.trim(),
   317	      tags: selectedTags,
   318	      reactions: { like: 0, dislike: 0, love: 0 },
   319	      timestamp: new Date().toISOString(),
   320	    };
   321	
   322	    const updatedComments = [commentData, ...comments];
   323	    setComments(updatedComments);
   324	    localStorage.setItem(`comments_${itemId}`, JSON.stringify(updatedComments));
   325	
   326	    try {
   327	      await fetch(API_ENDPOINT, {
   328	        method: 'POST',
   329	        body: JSON.stringify({ action: 'addComment', comment: commentData }),
   330	      });
   331	    } catch (error) {
   332	      console.warn("Failed to save to Google Sheet, saved locally.");
   333	    }
   334	
   335	    setNewComment('');
   336	    setSelectedTags([]);
   337	    setLoading(false);
   338	  };
   339	
   340	  return (
   341	    <div className="mt-8 border-t border-slate-200 dark:border-slate-800 pt-8">
   342	      <h3 className="text-xl font-black uppercase tracking-wider text-slate-900 dark:text-white mb-6 flex items-center gap-2">
   343	        <Icon name="MessageSquare" size={24} className="text-blue-500" /> 
   344	        Intel Reviews & Chatter
   345	      </h3>
   346	
   347	      {/* Add Comment Form */}
   348	      <form onSubmit={handleSubmit} className="bg-slate-50 dark:bg-slate-900/50 rounded-2xl p-4 sm:p-6 border border-slate-200 dark:border-slate-800 mb-8">
   349	        <div className="flex gap-4 mb-4">
   350	          <div className="w-10 h-10 rounded-full bg-slate-200 dark:bg-slate-800 flex items-center justify-center shrink-0 border border-slate-300 dark:border-slate-700">
   351	            <SiWolframlanguage size={20} className="text-slate-700 dark:text-slate-300" />
   352	          </div>
   353	          <div className="flex-1">
   354	            <input 
   355	              type="text" 
   356	              placeholder="Your Alias (Optional)..." 
   357	              value={authorName}
   358	              onChange={(e) => setAuthorName(e.target.value)}
   359	              className="w-full bg-transparent border-b border-slate-300 dark:border-slate-700 pb-2 text-sm font-bold text-slate-900 dark:text-white outline-none focus:border-blue-500 transition-colors mb-4 placeholder:text-slate-400"
   360	              maxLength={20}
   361	            />
   362	            <textarea
   363	              placeholder="Drop your intel, review, or funny thoughts here..."
   364	              value={newComment}
   365	              onChange={(e) => setNewComment(e.target.value)}
   366	              required
   367	              rows={3}
   368	              className="w-full bg-white dark:bg-slate-950 border border-slate-200 dark:border-slate-800 rounded-xl p-3 text-sm text-slate-900 dark:text-white outline-none focus:border-blue-500 transition-colors resize-none"
   369	            />
   370	          </div>
   371	        </div>
   372	
   373	        {/* Tags Selection */}
   374	        <div className="mb-4">
   375	          <p className="text-xs font-bold text-slate-500 uppercase tracking-wider mb-2">Tag this Intel (Max 3):</p>
   376	          <div className="flex flex-wrap gap-2">
   377	            {AVAILABLE_TAGS.map(tag => (
   378	              <button
   379	                key={tag}
   380	                type="button"
   381	                onClick={() => toggleTag(tag)}
   382	                className={`px-3 py-1.5 rounded-full text-[10px] sm:text-xs font-bold transition-all ${
   383	                  selectedTags.includes(tag)
   384	                    ? 'bg-blue-500 text-white shadow-md shadow-blue-500/20 scale-105'
   385	                    : 'bg-slate-200 dark:bg-slate-800 text-slate-600 dark:text-slate-400 hover:bg-slate-300 dark:hover:bg-slate-700'
   386	                }`}
   387	              >
   388	                {tag}
   389	              </button>
   390	            ))}
   391	          </div>
   392	        </div>
   393	
   394	        <div className="flex justify-end">
   395	          <button 
   396	            type="submit" 
   397	            disabled={loading || !newComment.trim()}
   398	            className="px-6 py-2.5 bg-blue-600 hover:bg-blue-500 text-white rounded-xl font-bold text-sm uppercase tracking-widest shadow-lg shadow-blue-500/20 transition-all active:scale-95 disabled:opacity-50 disabled:cursor-not-allowed flex items-center gap-2"
   399	          >
   400	            {loading ? <Icon name="Loader" size={16} className="animate-spin" /> : <Icon name="Send" size={16} />}
   401	            Post Intel
   402	          </button>
   403	        </div>
   404	      </form>
   405	
   406	      {/* Comments List */}
   407	      <div className="space-y-4">
   408	        {fetching ? (
   409	          <div className="flex justify-center py-8">
   410	            <Icon name="Loader" size={24} className="text-blue-500 animate-spin" />
   411	          </div>
   412	        ) : comments.length === 0 ? (
   413	          <div className="text-center py-10 bg-slate-50 dark:bg-slate-900/30 rounded-2xl border border-dashed border-slate-300 dark:border-slate-800">
   414	            <SiWolframlanguage size={40} className="mx-auto text-slate-300 dark:text-slate-700 mb-3 opacity-50" />
   415	            <p className="text-sm font-bold text-slate-500">No intel dropped yet. Be the first wolf to howl!</p>
   416	          </div>
   417	        ) : (
   418	          <AnimatePresence>
   419	            {comments.map((comment) => (
   420	              <motion.div 
   421	                key={comment.id}
   422	                initial={{ opacity: 0, y: 10 }}
   423	                animate={{ opacity: 1, y: 0 }}
   424	                className="bg-white dark:bg-slate-900 border border-slate-200 dark:border-slate-800 rounded-2xl p-4 sm:p-5 shadow-sm"
   425	              >
   426	                <div className="flex gap-3 sm:gap-4">
   427	                  <div className="w-10 h-10 rounded-full bg-gradient-to-br from-slate-100 to-slate-200 dark:from-slate-800 dark:to-slate-900 flex items-center justify-center shrink-0 border border-slate-200 dark:border-slate-700 shadow-inner">
   428	                    <SiWolframlanguage size={20} className="text-slate-800 dark:text-slate-200" />
   429	                  </div>
   430	                  <div className="flex-1 min-w-0">
   431	                    <div className="flex flex-wrap items-baseline justify-between gap-2 mb-1">
   432	                      <h4 className="font-black text-sm text-slate-900 dark:text-white truncate">
   433	                        {comment.author}
   434	                      </h4>
   435	                      <span className="text-[10px] font-bold text-slate-400 uppercase tracking-wider">
   436	                        {new Date(comment.timestamp).toLocaleDateString()}
   437	                      </span>
   438	                    </div>
   439	                    
   440	                    {comment.tags && comment.tags.length > 0 && (
   441	                      <div className="flex flex-wrap gap-1.5 mb-2">
   442	                        {comment.tags.map(tag => (
   443	                          <span key={tag} className="px-2 py-0.5 bg-slate-100 dark:bg-slate-800 text-slate-600 dark:text-slate-300 rounded-md text-[9px] font-bold uppercase tracking-wider border border-slate-200 dark:border-slate-700">
   444	                            {tag}
   445	                          </span>
   446	                        ))}
   447	                      </div>
   448	                    )}
   449	
   450	                    <p className="text-sm text-slate-700 dark:text-slate-300 whitespace-pre-wrap break-words leading-relaxed mb-3">
   451	                      {comment.text}
   452	                    </p>
   453	
   454	                    {/* Reactions */}
   455	                    <div className="flex items-center gap-2 sm:gap-3">
   456	                      <button 
   457	                        onClick={() => handleReaction(comment.id, 'like')}
   458	                        className="flex items-center gap-1.5 px-2.5 py-1 bg-slate-50 dark:bg-slate-800/50 hover:bg-slate-100 dark:hover:bg-slate-800 rounded-lg text-xs font-bold text-slate-600 dark:text-slate-400 transition-colors border border-slate-200 dark:border-slate-700/50"
   459	                      >
   460	                        👍 <span>{comment.reactions?.like || 0}</span>
   461	                      </button>
   462	                      <button 
   463	                        onClick={() => handleReaction(comment.id, 'dislike')}
   464	                        className="flex items-center gap-1.5 px-2.5 py-1 bg-slate-50 dark:bg-slate-800/50 hover:bg-slate-100 dark:hover:bg-slate-800 rounded-lg text-xs font-bold text-slate-600 dark:text-slate-400 transition-colors border border-slate-200 dark:border-slate-700/50"
   465	                      >
   466	                        👎 <span>{comment.reactions?.dislike || 0}</span>
   467	                      </button>
   468	                      <button 
   469	                        onClick={() => handleReaction(comment.id, 'love')}
   470	                        className="flex items-center gap-1.5 px-2.5 py-1 bg-rose-50 dark:bg-rose-500/10 hover:bg-rose-100 dark:hover:bg-rose-500/20 rounded-lg text-xs font-bold text-rose-600 dark:text-rose-400 transition-colors border border-rose-200 dark:border-rose-500/20"
   471	                      >
   472	                        ❤️ <span>{comment.reactions?.love || 0}</span>
   473	                      </button>
   474	                    </div>
   475	                  </div>
   476	                </div>
   477	              </motion.div>
   478	            ))}
   479	          </AnimatePresence>
   480	        )}
   481	      </div>
   482	    </div>
   483	  );
   484	};
